127  template <class I>
128  void operator=(I state_code)
129  {
130  if constexpr (std::is_same_v<I, std::error_code>)
131  {
132  *(throw_if_api_error*)this = state_code;
133  }
134  else if constexpr (std::is_same_v<I, BOOL>)
135  {
136  if (!state_code)
137  {
138  const_cast<std::error_code&>(this->code()).assign(GetLastError(), this->code().category());
139  throw *this;
140  }
141  }
142  else if constexpr (std::is_same_v<I, HANDLE>)
143  {
144  if (state_code == INVALID_HANDLE_VALUE)
145  {
146  const_cast<std::error_code&>(this->code()).assign(GetLastError(), this->code().category());
147  throw *this;
148  }
149  }
150  else if constexpr (std::is_same_v<I, DWORD>)
151  {
152  if (state_code != ERROR_SUCCESS)
153  {
154  const_cast<std::error_code&>(this->code()).assign(state_code, this->code().category());
155  throw *this;
156  }
157  }
158  else if constexpr (std::is_same_v<I, HRESULT>)
159  {
160  if (FAILED(state_code))
161  {
162  const_cast<std::error_code&>(this->code()).assign(state_code, hresult_category());
163  throw *this;
